#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* Reference C implementation of a tiny Huffman encoder used as an additional
 * test vector. The file is intentionally full of repeated identifiers,
 * punctuation and whitespace so that a dictionary-based coder such as LZW
 * has plenty of structure to exploit, whereas a naive Huffman coder has to
 * pay for every single character. */

typedef struct Node {
    int freq;
    int symbol;
    struct Node *left, *right;
} Node;

static Node *new_node(int freq, int symbol, Node *left, Node *right) {
    Node *n = (Node *) malloc(sizeof(Node));
    n->freq = freq;
    n->symbol = symbol;
    n->left = left;
    n->right = right;
    return n;
}

static int cmp(const void *a, const void *b) {
    const Node *na = *(const Node **) a;
    const Node *nb = *(const Node **) b;
    return na->freq - nb->freq;
}

static Node *build_tree(const int *freqs, int alphabet) {
    Node **heap = (Node **) malloc(sizeof(Node *) * alphabet);
    int n = 0;
    for (int i = 0; i < alphabet; ++i) {
        if (freqs[i] > 0) {
            heap[n++] = new_node(freqs[i], i, NULL, NULL);
        }
    }
    while (n > 1) {
        qsort(heap, n, sizeof(Node *), cmp);
        Node *a = heap[0];
        Node *b = heap[1];
        Node *merged = new_node(a->freq + b->freq, -1, a, b);
        heap[0] = merged;
        for (int i = 1; i < n - 1; ++i) heap[i] = heap[i + 1];
        n -= 1;
    }
    Node *root = heap[0];
    free(heap);
    return root;
}

int main(int argc, char **argv) {
    if (argc < 2) {
        fprintf(stderr, "usage: %s <text>\n", argv[0]);
        return 1;
    }
    int freqs[256] = {0};
    for (const char *p = argv[1]; *p; ++p) freqs[(unsigned char) *p] += 1;
    Node *root = build_tree(freqs, 256);
    printf("root freq = %d\n", root->freq);
    return 0;
}
